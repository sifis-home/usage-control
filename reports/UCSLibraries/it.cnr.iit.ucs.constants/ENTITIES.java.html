package it.cnr.iit.ucs.constants;

/**
 * Enumerate to handle default source and destination
 *
 * @author Antonio La Marra
 */
public enum ENTITIES {
    PIP( "PIP" ),
    CH( "CH" ),
    PEP( "PEP" );

    private String value;

    ENTITIES( String value ) {
        this.value = value;
    }

    @Override
    public String toString() {
        return value;
    }

}
